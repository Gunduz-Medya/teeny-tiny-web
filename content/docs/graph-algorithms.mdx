## ðŸ”¹ 2. Depth-First Search (DFS)

DFS explores a graph **recursively** or **using a stack**, visiting each node **deeply** before backtracking.

### **Example: Recursive DFS**
```javascript
function dfs(graph, node, visited = new Set()) {
  if (visited.has(node)) return;
  
  console.log(node); // Process node
  visited.add(node);
  
  for (let neighbor of graph[node]) {
    dfs(graph, neighbor, visited);
  }
}

dfs(graph, "A");
```

âœ… **DFS is useful for cycle detection and pathfinding.**

## ðŸ”¹ 3. Breadth-First Search (BFS)

BFS explores a graph **level by level** using a **queue**.

### **Example: BFS Implementation**
```javascript
function bfs(graph, start) {
  let queue = [start];
  let visited = new Set(queue);

  while (queue.length) {
    let node = queue.shift();
    console.log(node); // Process node

    for (let neighbor of graph[node]) {
      if (!visited.has(neighbor)) {
        queue.push(neighbor);
        visited.add(neighbor);
      }
    }
  }
}

bfs(graph, "A");
```

âœ… **BFS is used for shortest path and connectivity checks.**

## ðŸ”¹ 4. Dijkstraâ€™s Algorithm (Shortest Path)

Dijkstraâ€™s Algorithm finds the **shortest path from a source node** in a weighted graph.

### **Example: Dijkstraâ€™s Algorithm**
```javascript
function dijkstra(graph, start) {
  let distances = {};
  let visited = new Set();
  let pq = [[start, 0]];

  for (let node in graph) distances[node] = Infinity;
  distances[start] = 0;

  while (pq.length) {
    pq.sort((a, b) => a[1] - b[1]); // Sort by distance
    let [node, dist] = pq.shift();

    if (visited.has(node)) continue;
    visited.add(node);

    for (let [neighbor, weight] of Object.entries(graph[node])) {
      let newDist = dist + weight;
      if (newDist < distances[neighbor]) {
        distances[neighbor] = newDist;
        pq.push([neighbor, newDist]);
      }
    }
  }

  return distances;
}

const weightedGraph = {
  A: { B: 4, C: 2 },
  B: { A: 4, C: 5, D: 10 },
  C: { A: 2, B: 5, D: 3 },
  D: { B: 10, C: 3 }
};

console.log(dijkstra(weightedGraph, "A"));
```

âœ… **Dijkstraâ€™s algorithm is widely used in GPS navigation and network routing.**

## ðŸ”¹ 5. Topological Sorting (DAGs)

Topological sorting orders a **Directed Acyclic Graph (DAG)** linearly.

### **Example: Topological Sort**
```javascript
function topologicalSort(graph) {
  let visited = new Set();
  let stack = [];

  function dfs(node) {
    if (visited.has(node)) return;
    visited.add(node);

    for (let neighbor of graph[node]) dfs(neighbor);
    
    stack.push(node);
  }

  for (let node in graph) dfs(node);
  
  return stack.reverse();
}

const dag = {
  A: ["C"],
  B: ["C", "D"],
  C: ["E"],
  D: ["F"],
  E: ["F"],
  F: []
};

console.log(topologicalSort(dag));
```

âœ… **Topological sorting is used in task scheduling and dependency resolution.**
