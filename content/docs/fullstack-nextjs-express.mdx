## ðŸ”¹ 2. Setting Up the Express Server

Create an `server.js` file in the project root:

ðŸ“Œ **`server.js`**
```javascript
const express = require("express");
const next = require("next");
require("dotenv").config();

const dev = process.env.NODE_ENV !== "production";
const app = next({ dev });
const handle = app.getRequestHandler();

const server = express();

server.use(express.json()); // Parse JSON requests
server.use(require("cors")()); // Enable CORS

server.get("/api/hello", (req, res) => {
  res.json({ message: "Hello from Express API!" });
});

server.all("*", (req, res) => {
  return handle(req, res); // Pass everything else to Next.js
});

app.prepare().then(() => {
  server.listen(3001, () => {
    console.log("Server running on http://localhost:3001");
  });
});
```

âœ… **Now, Next.js and Express will run together!**

Run the app:

```bash
node server.js
```

## ðŸ”¹ 3. Connecting Next.js to Express API

Modify `next.config.js` to **proxy API requests** to Express:

ðŸ“Œ **`next.config.js`**
```javascript
module.exports = {
  async rewrites() {
    return [
      {
        source: "/api/:path*",
        destination: "http://localhost:3001/api/:path*",
      },
    ];
  },
};
```

âœ… **Now, requests to `/api/...` in Next.js will be forwarded to Express.**

## ðŸ”¹ 4. Fetching Data from Express API in Next.js

ðŸ“Œ **`pages/index.js`**
```javascript

export default function Home() {
  const [data, setData] = useState(null);

  useEffect(() => {
    fetch("/api/hello")
      .then((res) => res.json())
      .then((data) => setData(data.message));
  }, []);

  return <h1>{data || "Loading..."}</h1>;
}
```

## ðŸ”¹ 5. Adding MongoDB for Full-Stack Data Handling

### **5.1 Install Mongoose**
```bash
npm install mongoose
```

### **5.2 Connect to MongoDB in Express API**

ðŸ“Œ **`server.js`**
```javascript
const mongoose = require("mongoose");

mongoose.connect(process.env.MONGO_URI, {
  useNewUrlParser: true,
  useUnifiedTopology: true,
});

const UserSchema = new mongoose.Schema({ name: String });
const User = mongoose.model("User", UserSchema);

server.post("/api/users", async (req, res) => {
  const user = await User.create({ name: req.body.name });
  res.json(user);
});
```

### **5.3 Calling the API from Next.js**

ðŸ“Œ **`components/UserForm.js`**
```javascript
import { useState } from "react";

export default function UserForm() {
  const [name, setName] = useState("");

  const handleSubmit = async (e) => {
    e.preventDefault();
    const res = await fetch("/api/users", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ name }),
    });
    const data = await res.json();
    console.log("User added:", data);
  };

  return (
    <form onSubmit={handleSubmit}>
      <input value={name} onChange={(e) => setName(e.target.value)} />
      <button type="submit">Add User</button>
    </form>
  );
}
```

âœ… **Now, users can be added to MongoDB from Next.js!**
